
```applescript
use AppleScript version "2.4" -- Yosemite (10.10) or lateruse scripting additionsuse framework "Foundation"tell application "Keyboard Maestro"	set _macroUID to (the clipboard as text)	set _groupUID to my GetKMGroupUIDForMacroUID(_macroUID)	editMacro _groupUID	delay 1	editMacro _macroUIDend tellon KM_getmacros_asList()	tell application "Keyboard Maestro Engine"		set _plistText to getmacros with asstring	end tell		set _plistString to current application's NSString's stringWithString:_plistText	set _data to _plistString's dataUsingEncoding:(current application's NSUTF8StringEncoding)	set {_object, _error} to current application's NSPropertyListSerialization's propertyListWithData:_data options:0 format:(missing value) |error|:(reference)	return _object as listend KM_getmacros_asListon GetKMGroupUIDForMacroUID(macroUID)	set _plist to KM_getmacros_asList()	repeat with _group in _plist		repeat with _macro in _group's macros			if uid of _macro is macroUID then				return uid of _group			end if		end repeat	end repeat	return ""end GetKMGroupUIDForMacroUID
```
